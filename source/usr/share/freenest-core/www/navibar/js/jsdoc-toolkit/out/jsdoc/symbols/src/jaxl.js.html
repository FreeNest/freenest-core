 51    /** Poll rate in milliseconds */
 52    var pollRate = 500;
 53    
 54    /**@public
 55    Address of webchat php */
 56    var pollUrl = false;
 57    
 58    var jid = false;
 59    var polling = 0;
 60    var lastPoll = false;
 61    var connected = false;
 62    var disconnecting = false;
 63    var payloadHandler = false;
 64    var date = new Date();
 65    var payloadQueue = new Array();
 66    
 67    
 68    return {
 69       connect:connect,
 70       disconnect:disconnect,
 71       ping:ping,
 72       urlDecode:urlDecode,
 73       urlEncode:urlEncode
 74    }

 76    /** Connects to server */
 77    function connect (payload) {
 78       data.jaxl = 'connect';
 79       sendPayLoad(payload);
 80    }
 81 
 82    /** Disconnects from server */
 83    function disconnect() {
 84       data.jaxl = 'disconnect';
 85       sendPayLoad(data);
 86    }
 87 
 88    /** Pings server */
 89    function ping (payload) { //check if data exists..?
 90       data.jaxl = 'ping';
 91       sendPayLoad(payload);
 92    }
 93 
 94    /** Adds payload to queue, sends payload via ajax */
 95    function sendPayLoad (payload) {
 96 
 97       var now = date.getTime();
 98       payloadQueue.push(payload);
 99 
100       if(sinceLastPoll>pollRate) {
101          xhtmlRequest(payloadQueue.shift());
102       }
103 
104    }
105    /** Sends xhtmlRequest to {@link pollUrl}  */
106    function xhtmlRequest (data) {
107 
108       if(polling>0 && connected) { //and not pinging..?
109 
110 
111          $.ajax({
112             type: 'POST',
113             url: pollUrl,
114             data: data; //serializearray..?
115             beforeSend: function () {
116                lastPoll = date.getTime();
117                if(data['jaxl'] == 'disconnect']) {
118                   disconnecting = true;
119                }
120                ++polling;
121             },
122             success: function (payload) {
123                --polling;
124                handlePayload(payload);
125 
126             }
127 
128          });
129       }
130    }
131    /**Handles the payload from php. Empty payload pings back to php,
132       if payload.jaxl = 'jaxl', calls function in php,
133       otherwise calls UI's payload handler with payload as argument
134     */
135    function handlePayload(payload) {
136 
137       if(payload.length == 0) {
138          ping();
139       }
140       else {
141 
142          for(key in payload) {
143             if(key == null ) {
144                ping();
145             }
146             else if(payload[key].jaxl = 'jaxl') {
147                xhtmlRequest(payload[key]);
148             }
149             else {
150                call_user_func(jaxl.payloadHandler, payload[key]);
151             }
152          } //for
153       }
154 
155    }
